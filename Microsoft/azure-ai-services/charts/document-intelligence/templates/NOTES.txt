⚠️  IMPORTANT: License Download Mode Enabled
------------------------------------------------
The container is configured to download a license file from Azure.
This should only be done ONCE during initial setup.

After the license is downloaded:
1. Set documentIntelligence.azure.downloadLicense to false in values.yaml
2. Upgrade the release to run in disconnected mode

✓ Running in Disconnected Mode
-------------------------------
The container is running without Azure connectivity.
License and output logs are persisted to PVCs.

Access the Document Intelligence API:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "azure-ai-services.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "API Endpoint: http://$NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "azure-ai-services.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "API Endpoint: http://$SERVICE_IP:{{ .Values.service.port }}"
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "azure-ai-services.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "API Endpoint: http://127.0.0.1:5000"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 5000:5000
{{- end }}
